def initialize window, text = "Let's shut those little brats up!"
  @image = Gosu::Image.from_text window,
                                 "#{text}\nPress CTRL+Q to exit",
                                 Gosu::default_font_name,
                                 100
  @color = 0xffffff00
  @x, @y = 10, 10
end
